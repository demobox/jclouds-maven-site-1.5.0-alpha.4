503   /**
504    * this will build any context supported by the ApiMetadata. This includes the base
505    * {@link ApiMetadata#getContext() context}, or any {@link ApiMetadata#getWrappers() wrapper} it
506    * supports.
507    * 
508    * ex. {@code builder.build(BlobStoreContext.class) } will work, if {@code TypeToken<BlobStore>}
509    * is a configured {@link ApiMetadata#getWrappers() wrapper} of this api.
510    * 
511    * 
512    */
513   @SuppressWarnings("unchecked")
514   public <C extends Closeable> C build(final TypeToken<C> contextType) {
515      TypeToken<C> returnType = null;
516      if (contextType.isAssignableFrom(apiMetadata.getContext()))
517         returnType = (TypeToken<C>) apiMetadata.getContext();
518      else
519         try {
520            returnType = (TypeToken<C>) Apis.findWrapper(apiMetadata, contextType);
521         } catch (NoSuchElementException e) {
522            throw new IllegalArgumentException(String.format(
523                     "api %s not assignable from or transformable to %s; context: %s, wrappers: %s", apiMetadata,
524                     contextType, apiMetadata.getContext(), apiMetadata.getWrappers()));
525         }
526      return (C) buildInjector().getInstance(Key.get(TypeLiteral.get(returnType.getType())));
527   }
